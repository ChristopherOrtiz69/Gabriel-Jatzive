📋 Instrucciones: conectar con Google Sheets
1

Abre sheets.google.com y crea una hoja nueva. Nómbrala como quieras (ej: Invitados Boda).

2

En el menú de la hoja: Extensiones → Apps Script. Se abre el editor de código.

3

Borra todo el código que aparece y pega esto:

const HOJA = 'Invitados'; function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(HOJA); if (!sheet) { sheet = ss.insertSheet(HOJA); sheet.appendRow(['Nombre','Lugares','Confirmado','Fecha','ID']); sheet.getRange(1,1,1,5).setFontWeight('bold'); } if (data.action === 'add') { sheet.appendRow([ data.nombre, data.cupos, 'No ⬜', Utilities.formatDate(new Date(),'America/Mexico_City','dd/MM/yyyy HH:mm'), data.id ]); } else if (data.action === 'confirm') { var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][4]) === String(data.id)) { sheet.getRange(i+1, 3).setValue(data.confirmado ? 'Sí ✅' : 'No ⬜'); break; } } } else if (data.action === 'delete') { var rows = sheet.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (String(rows[i][4]) === String(data.id)) { sheet.deleteRow(i+1); break; } } } return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false,error:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
4

Haz clic en Implementar → Nueva implementación.
· Tipo: Aplicación web
· Ejecutar como: Yo
· Quién tiene acceso: Cualquier persona
Haz clic en Implementar y copia la URL que aparece.

5

Pega esa URL en el campo "Conexión con Google Sheets" arriba y haz clic en Guardar. ¡Listo!
